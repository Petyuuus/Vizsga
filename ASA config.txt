ASA config:
! --- Interfészek beállítása ---
interface e0/0
 nameif inside
 security-level 100
 ip address 192.168.70.1 255.255.255.0
 no shutdown
interface E0/1
 nameif outside
 security-level 0
 ip address 10.0.2.2 255.255.255.252
 no shutdown

! --- Objektumok a könnyebb kezelhetőségért ---
object network KORHAZ_LAN
 subnet 192.168.50.0 255.255.255.0

object network BACKUP_LAN
 subnet 192.168.70.0 255.255.255.0

! --- Biztonsági szabály (ACL) ---
! Csak a kórházi hálózatot engedjük be a backupra
access-list OUTSIDE_IN extended permit ip object KORHAZ_LAN object BACKUP_LAN
access-group OUTSIDE_IN in interface outside

! --- NAT mentesítés (Hogy az eredeti IP-k látszódjanak) ---
nat (inside,outside) source static BACKUP_LAN BACKUP_LAN destination static KORHAZ_LAN KORHAZ_LAN

! --- Alapértelmezett útvonal az R7 felé ---
route outside 0.0.0.0 0.0.0.0 10.0.2.1 1

! --- ICMP (Ping) engedélyezése teszthez ---
policy-map global_policy
 class inspection_default
  inspect icmp

! Létrehozzuk a hiányzó class-map-et:
class-map inspection_default
 match default-inspection-traffic
exit

! Hozzáadjuk a policy-map-hez az ICMP-t:
policy-map global_policy
 class inspection_default
  inspect icmp
exit

! Végül aktiváljuk az egészet globálisan az ASA-n:
service-policy global_policy global

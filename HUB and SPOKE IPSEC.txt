HUB and SPOKE IPSEC:

R3:
enable
configure terminal
! --- Phase 1: Kulcsok ---
crypto isakmp policy 1
 authentication pre-share
 exit
crypto isakmp key KORHAZVPN address 10.0.1.2
crypto isakmp key KORHAZVPN address 10.0.1.6

! --- Phase 2: Titkositasi eljaras ---
crypto ipsec transform-set VPN-SET esp-aes esp-sha-hmac
exit

! --- Forgalom kivalasztasa (GRE forgalom) ---
ip access-list extended VPN-TRAFFIC-R4
 permit ip host 10.0.1.10 host 10.0.1.2
 exit
ip access-list extended VPN-TRAFFIC-R5
 permit ip host 10.0.1.10 host 10.0.1.6
 exit

! --- Crypto Map beallitasa ---
crypto map KORHAZ-MAP 10 ipsec-isakmp
 set peer 10.0.1.2
 set transform-set VPN-SET
 match address VPN-TRAFFIC-R4
 exit
crypto map KORHAZ-MAP 20 ipsec-isakmp
 set peer 10.0.1.6
 set transform-set VPN-SET
 match address VPN-TRAFFIC-R5
 exit

! --- Map alkalmazasa a fizikai WAN (ISP) labon ---
interface gig0/2/0
 crypto map KORHAZ-MAP
 exit

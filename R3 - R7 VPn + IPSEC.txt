R3 - R7 VPn + IPSEC
R3:
en
conf t
crypto isakmp policy 10
 authentication pre-share
exit
crypto isakmp key VPN_KULCS address 10.0.1.14
crypto ipsec transform-set VPN_SET esp-aes esp-sha-hmac
exit

! --- Érdekes forgalom (Kórház -> Backup) ---
ip access-list extended VPN-BACKUP 
permit ip 192.168.50.0 0.0.0.255 192.168.70.0 0.0.0.255

! --- Crypto Map ---
crypto map VPN_MAP 10 ipsec-isakmp
 set peer 10.0.1.14
 set transform-set VPN_SET
 match address VPN-BACKUP
exit

! --- Alkalmazás a külső lábon (ISP4 felé) ---
interface GigabitEthernet0/2/0 
 crypto map VPN_MAP
exit

! --- Útvonal a Backup szerverek felé (ha még nincs) ---
ip route 192.168.70.0 255.255.255.0 10.0.1.9  ! Az ISP4 router felé
